import os
import json
import pandas as pd
import re
from typing import List, Dict, Tuple, Optional
from collections import defaultdict

def load_config_from_file(config_file: str) -> Dict:
    try:
        with open(config_file, 'r') as f:
            return json.load(f)
    except Exception as e:
        print(f"Error loading config file: {e}")
        return {}

class DatabricksSchemaGenerator:
    def __init__(self,
                 excel_file_path: str,
                 output_base_folder: str = "generated_schemas",
                 sheet_name: Optional[str] = None,
                 categories: Optional[Dict[str, Dict[str, str]]] = None,
                 category_list: Optional[List[str]] = None,
                 default_datatype_col_suffix: str = " - Data Type"):
        self.excel_file_path = excel_file_path
        self.output_base_folder = output_base_folder
        self.sheet_name = sheet_name
        self.default_datatype_col_suffix = default_datatype_col_suffix
        
        # If category_list is provided, auto-generate categories config
        if category_list:
            self.categories_config = self._generate_categories_config(category_list)
        else:
            self.categories_config = categories or {}
        
        self.tables_created = 0

    def _generate_categories_config(self, category_list: List[str]) -> Dict[str, Dict[str, str]]:
        """Auto-generate categories config based on category names and column patterns"""
        config = {}
        
        for category in category_list:
            config[category] = {
                "schema_col": f"{category} - Schema",
                "table_col": f"{category} - Physical Table Name", 
                "column_col": f"{category} - Physical Column Name",
                "datatype_col": f"{category}{self.default_datatype_col_suffix}"
            }
            
        return config

    def _find_matching_columns(self, df: pd.DataFrame, category: str) -> Dict[str, str]:
        """Find matching columns in the dataframe for a given category"""
        available_cols = df.columns.tolist()
        found_cols = {}
        
        # Expected column patterns
        patterns = {
            "schema_col": [f"{category} - Schema", f"{category}- Schema"],
            "table_col": [f"{category} - Physical Table Name", f"{category}- Physical Table Name"],
            "column_col": [f"{category} - Physical Column Name", f"{category}- Physical Column Name"],
            "datatype_col": [f"{category} - Data Type", f"{category}- Data Type", f"{category}{self.default_datatype_col_suffix}"]
        }
        
        for col_type, possible_names in patterns.items():
            for possible_name in possible_names:
                if possible_name in available_cols:
                    found_cols[col_type] = possible_name
                    break
        
        return found_cols

    def map_datatype(self, datatype: str) -> str:
        if pd.isna(datatype) or not datatype or str(datatype).strip() == '':
            return 'VARCHAR(255)'
        dtype = str(datatype).upper().strip()

        if m := re.match(r'VARCHAR2\((\d+)\s*BYTE\)', dtype):
            return f'VARCHAR({m.group(1)})'
        if m := re.match(r'VARCHAR2\((\d+)\)', dtype):
            return f'VARCHAR({m.group(1)})'
        if m := re.match(r'NUMBER\((\d+),\s*(\d+)\)', dtype):
            return f'DECIMAL({m.group(1)},{m.group(2)})'
        if m := re.match(r'NUMBER\((\d+)\)', dtype):
            return 'INT' if int(m.group(1)) <= 10 else 'BIGINT'
        if 'TIMESTAMP' in dtype:
            return 'TIMESTAMP'
        if 'DATE' in dtype:
            return 'DATE'
        if 'CHAR' in dtype:
            return 'VARCHAR(1)'
        if 'CLOB' in dtype:
            return 'STRING'
        if 'BLOB' in dtype:
            return 'BINARY'
        return 'VARCHAR(255)'

    def load_excel_data(self) -> pd.DataFrame:
        try:
            df = pd.read_excel(self.excel_file_path, sheet_name=self.sheet_name)
            
            # Print column names for debugging
            print("Available columns in Excel:")
            for col in df.columns:
                print(f"  - '{col}'")
            
            # Clean column names (remove extra spaces)
            df.columns = df.columns.str.strip()
            
            # Forward-fill missing schema/table names for each category
            for cat, cfg in self.categories_config.items():
                schema_col = cfg.get("schema_col")
                table_col = cfg.get("table_col")
                
                if schema_col and schema_col in df.columns:
                    df[schema_col] = df[schema_col].ffill()
                if table_col and table_col in df.columns:
                    df[table_col] = df[table_col].ffill()

            # Drop rows where any schema column is 'NA' or empty
            for cfg in self.categories_config.values():
                schema_col = cfg.get("schema_col")
                if schema_col and schema_col in df.columns:
                    df = df[
                        (df[schema_col].notna()) & 
                        (df[schema_col].astype(str).str.upper() != "NA") &
                        (df[schema_col].astype(str).str.strip() != "")
                    ]

            return df

        except Exception as e:
            print(f"Error loading Excel file: {e}")
            return pd.DataFrame()

    def extract_tables(self, df: pd.DataFrame) -> Dict[str, Dict[Tuple[str, str], List[Tuple[str, str]]]]:
        result = {}
        
        for cat in self.categories_config.keys():
            # Find matching columns in the dataframe
            found_cols = self._find_matching_columns(df, cat)
            
            schema_col = found_cols.get('schema_col')
            table_col = found_cols.get('table_col')
            column_col = found_cols.get('column_col')
            datatype_col = found_cols.get('datatype_col')  # This might be None

            print(f"\nProcessing category: {cat}")
            print(f"  Schema column: {schema_col}")
            print(f"  Table column: {table_col}")
            print(f"  Column column: {column_col}")
            print(f"  Datatype column: {datatype_col}")

            # Check if required columns exist
            missing_cols = []
            for col_name, col_val in [('schema_col', schema_col), ('table_col', table_col), ('column_col', column_col)]:
                if not col_val:
                    missing_cols.append(col_name)
            
            if missing_cols:
                print(f"  WARNING: Missing columns for {cat}: {missing_cols}")
                result[cat] = {}
                continue

            tables = defaultdict(list)
            seen = defaultdict(set)

            # Filter rows that have valid data for this category
            category_filter = (
                df[schema_col].notna() & 
                df[table_col].notna() & 
                df[column_col].notna() &
                (df[schema_col].astype(str).str.strip() != '') &
                (df[table_col].astype(str).str.strip() != '') &
                (df[column_col].astype(str).str.strip() != '')
            )
            
            filtered_df = df[category_filter]
            print(f"  Found {len(filtered_df)} valid rows for {cat}")

            for _, row in filtered_df.iterrows():
                schema = str(row[schema_col]).strip()
                table = str(row[table_col]).strip()
                column = str(row[column_col]).strip()
                
                # Handle datatype column (might not exist for all categories)
                if datatype_col and datatype_col in df.columns:
                    dtype_val = row.get(datatype_col, '')
                else:
                    dtype_val = ''
                    if not datatype_col:
                        print(f"  INFO: No datatype column found for {cat}, using default VARCHAR(255)")

                # Skip if any required field is empty
                if not schema or not table or not column:
                    continue

                # Check for duplicates (case-insensitive)
                if column.lower() not in seen[(schema, table)]:
                    dtype = self.map_datatype(dtype_val) if dtype_val else 'VARCHAR(255)'
                    tables[(schema, table)].append((column, dtype))
                    seen[(schema, table)].add(column.lower())

            print(f"  Extracted {len(tables)} tables for {cat}")
            for (schema, table), cols in tables.items():
                print(f"    {schema}.{table}: {len(cols)} columns")

            result[cat] = tables

        return result

    def generate_schema_sql(self, schema: str, table: str, columns: List[Tuple[str, str]], category: str) -> str:
        sql = f"-- {category} - {table} Table Schema\n"
        sql += f"CREATE TABLE IF NOT EXISTS external_catalog.{schema}.{table} (\n"
        sql += ",\n".join([f"    [{col}] {dtype}" for col, dtype in columns])
        sql += "\n);"
        return sql

    def generate_drop_sql(self, schema: str, table: str, category: str) -> str:
        sql = f"-- {category} - {table} Table Drop\n"
        sql += f"DROP TABLE IF EXISTS external_catalog.{schema}.{table};"
        return sql

    def create_folder_structure(self):
        folders = list(self.categories_config.keys()) + ["consolidated"]
        os.makedirs(self.output_base_folder, exist_ok=True)
        for folder in folders:
            # Create main category folder
            os.makedirs(os.path.join(self.output_base_folder, folder), exist_ok=True)
            # Create subfolders for create and drop scripts
            os.makedirs(os.path.join(self.output_base_folder, folder, "create"), exist_ok=True)
            os.makedirs(os.path.join(self.output_base_folder, folder, "drop"), exist_ok=True)
        
        # Create consolidated subfolders
        consolidated_path = os.path.join(self.output_base_folder, "consolidated")
        os.makedirs(os.path.join(consolidated_path, "create"), exist_ok=True)
        os.makedirs(os.path.join(consolidated_path, "drop"), exist_ok=True)

    def generate_category_consolidated_schema(self, category: str, tables: Dict[Tuple[str, str], List[Tuple[str, str]]]) -> Tuple[str, str, int]:
        create_sql = f"-- {category} CATEGORY - CONSOLIDATED CREATE SCHEMA\n\n"
        drop_sql = f"-- {category} CATEGORY - CONSOLIDATED DROP SCHEMA\n\n"
        count = 0
        
        for (schema, table), cols in tables.items():
            if cols:
                create_sql += self.generate_schema_sql(schema, table, cols, category) + "\n\n"
                drop_sql += self.generate_drop_sql(schema, table, category) + "\n\n"
                count += 1
                
        create_sql += f"-- Total tables in {category} consolidated create schema: {count}\n"
        drop_sql += f"-- Total tables in {category} consolidated drop schema: {count}\n"
        
        return create_sql, drop_sql, count

    def generate_master_consolidated_schema(self, all_tables: Dict[str, Dict[Tuple[str, str], List[Tuple[str, str]]]]) -> Tuple[str, str]:
        create_sql = "-- MASTER CONSOLIDATED CREATE SCHEMA\n\n"
        drop_sql = "-- MASTER CONSOLIDATED DROP SCHEMA\n\n"
        total = 0
        
        for cat, tables in all_tables.items():
            if tables:  # Only add category if it has tables
                create_sql += f"-- {cat} CATEGORY\n\n"
                drop_sql += f"-- {cat} CATEGORY\n\n"
                
                for (schema, table), cols in tables.items():
                    if cols:
                        create_sql += self.generate_schema_sql(schema, table, cols, cat) + "\n\n"
                        drop_sql += self.generate_drop_sql(schema, table, cat) + "\n\n"
                        total += 1
                        
        create_sql += f"-- Total tables in master consolidated create schema: {total}\n"
        drop_sql += f"-- Total tables in master consolidated drop schema: {total}\n"
        
        return create_sql, drop_sql

    def run(self):
        df = self.load_excel_data()
        if df.empty:
            print("No data loaded.")
            return

        print(f"Loaded {len(df)} rows from Excel file")
        
        self.create_folder_structure()
        all_tables = self.extract_tables(df)
        total_tables = 0

        for cat, tables in all_tables.items():
            create_folder = os.path.join(self.output_base_folder, cat, "create")
            drop_folder = os.path.join(self.output_base_folder, cat, "drop")
            count = 0
            
            # Generate individual table files
            for (schema, table), cols in tables.items():
                if cols:
                    # Generate CREATE script
                    create_sql = self.generate_schema_sql(schema, table, cols, cat)
                    create_filename = f"{schema}_{table}.sql".lower()
                    with open(os.path.join(create_folder, create_filename), "w") as f:
                        f.write(create_sql)
                    
                    # Generate DROP script
                    drop_sql = self.generate_drop_sql(schema, table, cat)
                    drop_filename = f"{schema}_{table}.sql".lower()
                    with open(os.path.join(drop_folder, drop_filename), "w") as f:
                        f.write(drop_sql)
                    
                    count += 1

            # Generate consolidated schemas for this category
            if tables:
                create_consolidated_sql, drop_consolidated_sql, cat_count = self.generate_category_consolidated_schema(cat, tables)
                
                # Write consolidated CREATE file
                with open(os.path.join(create_folder, f"{cat.lower()}_consolidated_create_{cat_count}.sql"), "w") as f:
                    f.write(create_consolidated_sql)
                
                # Write consolidated DROP file  
                with open(os.path.join(drop_folder, f"{cat.lower()}_consolidated_drop_{cat_count}.sql"), "w") as f:
                    f.write(drop_consolidated_sql)
                
                print(f"Created {count} individual CREATE/DROP table files and 1 consolidated CREATE/DROP file for {cat}")
            else:
                print(f"No tables found for category: {cat}")

            total_tables += count

        # Generate master consolidated schemas
        master_create_sql, master_drop_sql = self.generate_master_consolidated_schema(all_tables)
        
        # Write master consolidated CREATE file
        with open(os.path.join(self.output_base_folder, "consolidated", "create", "all_tables_master_consolidated_create.sql"), "w") as f:
            f.write(master_create_sql)
        
        # Write master consolidated DROP file
        with open(os.path.join(self.output_base_folder, "consolidated", "drop", "all_tables_master_consolidated_drop.sql"), "w") as f:
            f.write(master_drop_sql)

        print(f"\nTotal tables created: {total_tables}")
        print(f"Files generated in: {self.output_base_folder}")
